import streamlit as st
import pandas as pd
import pydeck as pdk
from geopy.geocoders import Nominatim
import plotly.express as px
import time

st.set_page_config(layout="wide")

st.title("🚚 Transportation Route Dashboard")

# Load Data
df = pd.read_csv("transport.csv")

# KPI Cards
col1, col2, col3 = st.columns(3)

col1.metric("Total Routes", len(df))
col2.metric("Total KM", int(df["KM"].sum()))
col3.metric("Longest Route", int(df["KM"].max()))

# Top Routes
st.subheader("Top 10 Longest Routes")

top_routes = df.sort_values(by="KM", ascending=False).head(10)

fig = px.bar(
    top_routes,
    x="KM",
    y="Source",
    color="KM",
    orientation="h",
    hover_data=["Destination"]
)

st.plotly_chart(fig, use_container_width=True)

# Geocoder
geolocator = Nominatim(user_agent="transport_app")

@st.cache_data
def get_coordinates(place):
    try:
        location = geolocator.geocode(place + ", India")
        time.sleep(1)
        return location.latitude, location.longitude
    except:
        return None, None

# Add Coordinates
source_lat = []
source_lon = []
dest_lat = []
dest_lon = []

for _, row in df.iterrows():
    slat, slon = get_coordinates(row["Source"])
    dlat, dlon = get_coordinates(row["Destination"])

    source_lat.append(slat)
    source_lon.append(slon)
    dest_lat.append(dlat)
    dest_lon.append(dlon)

df["source_lat"] = source_lat
df["source_lon"] = source_lon
df["dest_lat"] = dest_lat
df["dest_lon"] = dest_lon

# Remove missing locations
map_df = df.dropna()

# Arc Layer
arc_layer = pdk.Layer(
    "ArcLayer",
    data=map_df,
    get_source_position='[source_lon, source_lat]',
    get_target_position='[dest_lon, dest_lat]',
    get_width='KM / 500',
    get_source_color='[255, 0, 0]',
    get_target_color='[0, 255, 0]',
    pickable=True,
    auto_highlight=True
)

view_state = pdk.ViewState(
    latitude=22.5937,
    longitude=78.9629,
    zoom=4,
    pitch=30
)

st.subheader("India Transportation Routes")

r = pdk.Deck(
    layers=[arc_layer],
    initial_view_state=view_state,
    tooltip={
        "text": "{Source} → {Destination}\nKM: {KM}"
    }
)

st.pydeck_chart(r)
